```go
func buildCallInputFields(args []parser.Arg) string {
    var fields []string
    for _, a := range args {
        fields = append(fields, argToCode(a))  // unkeyed positional
    }

}

// 변경 후
func buildCallInputFields(args []parser.Arg) string {
    var fields []string
    for _, a := range args {
        name := callFieldName(a)
        value := argToCode(a)
        fields = append(fields, name+": "+value)
    }
    return strings.Join(fields, ", ")
}

func callFieldName(a parser.Arg) string {
    if a.Literal != "" { return ucFirst(a.Literal) }
    if a.Field != ""   { return a.Field }
    return ucFirst(a.Source)  // bare variable
}
```
